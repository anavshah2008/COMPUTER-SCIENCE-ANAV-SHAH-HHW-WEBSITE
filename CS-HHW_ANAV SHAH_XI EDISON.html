Code to check if the number entered is Prime or not: 
number = int(input("Enter a number to check if it is prime: "))
def is_prime(num):
    # If the number is less than or equal to 1, it is not prime
    if num <= 1:
        return False
    # Check for factors from 2 to num-1
    for i in range(2, num):
        if num % i == 0:
            return False
    return True
if is_prime(number):
    print(f"{number} is a prime number.")
else:
    print(f"{number} is not a prime number.")
Code to generate Fibonacci Series: 
terms = int(input("Enter the number of terms for Fibonacci series: "))
def generate_fibonacci(n):
    fib = [0, 1]  # Initialize the first two terms
    for i in range(2, n):
        fib.append(fib[i - 1] + fib[i - 2])  # Append the sum of the last two terms
    return fib
print("Fibonacci series:", generate_fibonacci(terms))
Code to check for Leap year: 
year = int(input("Enter a year to check if it is a leap year: "))
def is_leap_year(year):
    if year % 4 == 0:
        if year % 100 == 0:
            if year % 400 == 0:
                return True
            return False
        return True
    return False
if is_leap_year(year):
    print(f"{year} is a leap year.")
else:
    print(f"{year} is not a leap year.")
Code to check if the date entered is valid or not: 
from datetime import datetime

day = int(input("Enter day: "))
month = int(input("Enter month: "))
year = int(input("Enter year: "))
def is_valid_date(day, month, year):
    try:
        datetime(year, month, day)
        return True
    except ValueError:
        return False
if day == 0 and month == 0 and year == 0:
    print("Invalid date (0/0/0).")
else:
    if is_valid_date(day, month, year):
        print(f"The date {day}/{month}/{year} is valid.")
    else:
        print(f"The date {day}/{month}/{year} is invalid.")
Code to check if the number is Palindrome or not: 
number = int(input("Enter a number to check if it is a palindrome: "))
def is_palindrome(num):
    str_num = str(num)  # Convert the number to a string
    reversed_str_num = str_num[::-1]  # Reverse the string
    return str_num == reversed_str_num
if is_palindrome(number):
    print(f"{number} is a palindrome.")
else:
    print(f"{number} is not a palindrome.")
Code to take a 3 digit number and display all digits of the words of that number in a different line: 
number = int(input("Enter a 3-digit number: "))
def display_digits(num):
    if num < 100 or num > 999:
        return "Number is not a 3-digit number."
    digits = str(num)
    return '\n'.join(digits)  # Join the digits with newline characters
print(display_digits(number))
Code to accept numbers and show its digits:

def show_digits(num):
    return ', '.join(str(num))  # Join the digits with commas
print("Digits:", show_digits(number))
Code to generate a calculator: 
num1 = float(input("Enter the first number: "))
num2 = float(input("Enter the second number: "))
op = input("Enter an operator (+, -, *, /): ")
def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    if b == 0:
        return 'Error: Division by zero'
    return a / b

# Function to use the calculator
def calculator(a, b, operator):
    if operator == '+':
        return add(a, b)
    elif operator == '-':
        return subtract(a, b)
    elif operator == '*':
        return multiply(a, b)
    elif operator == '/':
        return divide(a, b)
    else:
        return 'Invalid operator'
result = calculator(num1, num2, op)
print(f"Result: {result}")
